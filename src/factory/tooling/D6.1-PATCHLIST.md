**File:** `src/factory/tooling/storage/indexedDbToolingStore.ts`

```typescript
// ADD: Reset tool wear record
export async function resetToolWear(
  toolId: string,
  note?: string
): Promise<void> {
  const db = await getToolingDb();
  const tx = db.transaction('toolUsageRecords', 'readwrite');

  // Delete existing record (starts fresh)
  await tx.store.delete(toolId);

  // Optionally log the reset (local audit)
  if (note) {
    await logToolMaintenanceAction({
      toolId,
      action: 'RESET',
      note,
      timestamp: Date.now(),
    });
  }

  await tx.done;
}

// ADD: New store for maintenance log (optional)
// Store: 'toolMaintenanceLog' (keyPath: 'id', autoIncrement)
interface ToolMaintenanceEntry {
  id?: number;
  toolId: string;
  action: 'RESET' | 'THRESHOLD_CHANGE';
  note?: string;
  timestamp: number;
}
```
